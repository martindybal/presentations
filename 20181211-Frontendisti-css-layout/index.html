Grid autoprefixer

  • Od verze 7.2. Doporučuji verzi >= 9.3.1
  • Je nutno zapnout
  • Pouze pro IE 10 a 11 a podpora není 100%
  • Přídá podporu
    • grid-gap
    • grid-template-areas

plugins: {
    autoprefixer: { 
        grid: true,
        browsers: "last 3 versions" 
    }
}

/* autoprefixer grid: on */
.grid-container {
    display: grid;
}

if(typeof CSS != "undefined") {
    if (CSS.supports && CSS.supports('display: grid')) {
        document.documentElement.className = 
            document.documentElement.className.replace("no-cssgrid", "cssgrid");
    }        
}
